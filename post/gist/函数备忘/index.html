#include <unistd.h>
#include <stdio.h>
#include <stdlib.h>
#include <signal.h>
void handler(int sig)
{
    printf("get a sig,num is %d\n", sig );
    exit(0) ;
}

int main(void)
{
	int counter;
	signal(SIGALRM,handler);
	alarm(1);
	for(counter=0; 1; counter++)
        printf("counter=%d \n", counter);
	return 0;
}

#include <iostream>  
#include <sys/types.h>  
#include <sys/stat.h>  
#include <sys/wait.h>  
#include <fcntl.h>  
#include <unistd.h>  
#include <stdlib.h>  
#include <stdio.h>  
#include <sys/inotify.h>  
using namespace std;  
  
void watch_inotify(void)  
{  
    int fd = 0;  
    char data[255] = {0};  
    int num = 0;  
  
    fd = inotify_init();//初始化fd  
    inotify_add_watch(fd, "lt", IN_OPEN);//监听文件是否打开  
    num = read(fd, data, 255);//没打开，那么就会阻塞  
    cout << num << endl;//如果打开，那么会继续跑到这里  
}  
int main()  
{  
    watch_inotify();  
    return 0;  
}
